Crear tablas

Sentencia:

CREATE TABLE table_name (
 column1 datatype key,
 column2 datatype,
 column3 datatype,
 ...);

EJEMPLO

CREATE TABLE customers (
 id INTEGER PRIMARY KEY,
 name TEXT,
 age INTEGER,
 weight REAL);
Insertar datos

Sentencia:

INSERT INTO table_name
 VALUES (value1, value2, ...);

INSERT INTO table_name (column1, ...
 VALUES (value1, ...);

EJEMPLO

INSERT INTO customers
 VALUES (73, "Briam", 33);

INSERT INTO customers (name, age)
 VALUES ("Briam", 33);
Consultar datos

Sentencia:

SELECT *
 FROM table_name;

SELECT column1, column2, ...
 FROM table_name;

EJEMPLO

SELECT *
 FROM customers;

SELECT name, age
 FROM customers;
Actualizar datos

Sentencia:

UPDATE table_name
 SET column1 = value1, ...
 WHERE condition;

EJEMPLO

UPDATE customers
 SET age = 33
 WHERE id = 73;
Eliminar datos

Sentencia:

DELETE FROM table_name
 WHERE condition;

EJEMPLO

DELETE FROM customers
 WHERE id = 73;
